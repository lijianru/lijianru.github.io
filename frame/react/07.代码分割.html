import() вҖӢ

еңЁеә”з”Ёдёӯеј•е…Ҙд»Јз ҒеҲҶеүІзҡ„жңҖдҪіж–№ејҸжҳҜйҖҡиҝҮеҠЁжҖҒ import()гҖӮеҪ“ webpack и§ЈжһҗеҲ°иҜҘиҜӯжі•зҡ„ж—¶еҖҷдјҡиҮӘеҠЁзҡ„ејҖе§Ӣд»Јз ҒеҲҶеүІгҖӮпјҲCreate-React-APP е’Ң Next.js е·Іж”ҜжҢҒиҜҘзү№жҖ§пјү

javascript
import('./math').then((math) => {
  console.log(math.add(16, 26));
});

React.lazy вҖӢ

React.lazy еҮҪж•°иғҪеғҸжёІжҹ“еёёи§„з»„д»¶дёҖж ·еӨ„зҗҶеҠЁжҖҒеј•е…Ҙзҡ„з»„д»¶гҖӮReact.lazy жҺҘеҸ—дёҖдёӘеҮҪж•°пјҢиҝҷдёӘеҮҪж•°йңҖиҰҒеҠЁжҖҒи°ғз”Ё import()гҖӮе®ғеҝ…йЎ»иҝ”еӣһдёҖдёӘ PromiseпјҢиҜҘ Promise йңҖиҰҒ resolve дёҖдёӘ default export зҡ„ React з»„д»¶гҖӮ

jsx
const OtherComponent = React.lazy(() => import('./OtherComponent'));

function MyComponent() {
  return (
    <div>
      <OtherComponent />
    </div>
  );
}

жіЁпјҡReact.lazy зӣ®еүҚеҸӘж”ҜжҢҒй»ҳи®ӨеҜјеҮәпјҲdefault exportsпјүпјҢеҸҜдёәжІЎжңүй»ҳи®ӨеҜјеҮәзҡ„еҲӣе»әдёҖдёӘдёӯй—ҙжЁЎеқ—

jsx
export { MyComponent as default } from './ManyComponents.js';

Suspense вҖӢ

еҰӮжһңеңЁ MyComponent жёІжҹ“е®ҢжҲҗеҗҺеҢ…еҗ« OtherComponent зҡ„жЁЎеқ—иҝҳжІЎжңүиў«еҠ иҪҪе®ҢжҲҗпјҢеҸҜдҪҝз”Ё Suspense з»„д»¶еҒҡйҷҚзә§еӨ„зҗҶгҖӮfallback еұһжҖ§жҺҘеҸ—д»»дҪ•еңЁз»„д»¶еҠ иҪҪиҝҮзЁӢдёӯдҪ жғіеұ•зӨәзҡ„ React е…ғзҙ 

jsx
const OtherComponent = React.lazy(() => import('./OtherComponent'));
function MyComponent() {
  return (
    <div>
      <Suspense fallback={<div>Loading...</div>}>
        <OtherComponent />
      </Suspense>
    </div>
  );
}

йҒҝе…Қе…ңеә• вҖӢ

еҪ“ react и§Ҷеӣҫд»ҺдёҖдёӘзҠ¶жҖҒеҲҮжҚўеҲ°еҸҰдёҖдёӘзҠ¶жҖҒж—¶пјҢеҰӮжһңдёӢдёҖдёӘзҠ¶жҖҒжІЎжңүеҮҶеӨҮеҘҪпјҢеҲҷдјҡдҪҝз”Ё Suspense дёӯзҡ„е…ңеә•ж–№жЎҲгҖӮдҪҶиҝҷеҸҜиғҪи®©з”ЁжҲ·ж„ҹеҲ°еӣ°жғ‘пјҢеҜ№дәҺиҝҷзұ»дёҚзҙ§жҖҘзҡ„жӣҙж–°жҲ‘д»¬еҸҜд»ҘдҪҝз”Ё transitions зӯүе®ғеҮҶеӨҮеҘҪеҗҺеңЁеҲҮжҚўгҖӮ

diff
import React, { Suspense } from 'react';
import Tabs from './Tabs';
import Glimmer from './Glimmer';

const Comments = React.lazy(() => import('./Comments'));
const Photos = React.lazy(() => import('./Photos'));

function MyComponent() {
  const [tab, setTab] = React.useState('photos');

  function handleTabSelect(tab) {
+    startTransition(() => {
      setTab(tab);
+    });
  };

  return (
    <div>
      <Tabs onTabSelect={handleTabSelect} />
      <Suspense fallback={<Glimmer />}>
        {tab === 'photos' ? <Photos /> : <Comments />}
      </Suspense>
    </div>
  );
}

еә”з”Ё вҖӢ

еҹәдәҺи·Ҝз”ұзҡ„д»Јз ҒеҲҶеүІ вҖӢ

jsx
import { BrowserRouter as Router, Route, Switch } from 'react-router-dom';
import React, { Suspense, lazy } from 'react';

const Home = lazy(() => import('./routes/Home'));
const About = lazy(() => import('./routes/About'));

const App = () => (
  <Router>
    <Suspense fallback={<div>Loading...</div>}>
      <Switch>
        <Route exact path="/" component={Home} />
        <Route path="/about" component={About} />
      </Switch>
    </Suspense>
  </Router>
);
